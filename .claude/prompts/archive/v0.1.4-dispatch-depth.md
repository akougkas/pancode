### 1. `src/domains/dispatch/validation.ts` (NEW)

Post-dispatch output validation. Optional per-dispatch contracts.

```typescript
import { existsSync } from "node:fs";
import { execSync } from "node:child_process";

export interface OutputContract {
  expectedFiles?: string[];
  expectedPatterns?: string[];         // regex patterns that must appear in worker output
  validationCommand?: string;          // shell command, exit 0 = pass
  validationTimeoutMs?: number;        // default 10000, max 30000
}

export interface ValidationResult {
  valid: boolean;
  checks: ValidationCheck[];
}

export interface ValidationCheck {
  kind: "file_exists" | "pattern_match" | "command";
  target: string;
  passed: boolean;
  detail?: string;
}

export function validateOutput(output: string, cwd: string, contract: OutputContract): ValidationResult {
  const checks: ValidationCheck[] = [];

  // File existence checks
  if (contract.expectedFiles) {
    for (const file of contract.expectedFiles) {
      const fullPath = file.startsWith("/") ? file : `${cwd}/${file}`;
      const passed = existsSync(fullPath);
      checks.push({
        kind: "file_exists",
        target: file,
        passed,
        detail: passed ? undefined : `File not found: ${fullPath}`,
      });
    }
  }

  // Pattern matching in output
  if (contract.expectedPatterns) {
    for (const pattern of contract.expectedPatterns) {
      try {
        const regex = new RegExp(pattern);
        const passed = regex.test(output);
        checks.push({
          kind: "pattern_match",
          target: pattern,
          passed,
          detail: passed ? undefined : `Pattern not found in output`,
        });
      } catch {
        checks.push({
          kind: "pattern_match",
          target: pattern,
          passed: false,
          detail: `Invalid regex: ${pattern}`,
        });
      }
    }
  }

  // Shell validation command
  if (contract.validationCommand) {
    const timeout = Math.min(contract.validationTimeoutMs ?? 10000, 30000);
    try {
      execSync(contract.validationCommand, { cwd, timeout, stdio: "pipe" });
      checks.push({ kind: "command", target: contract.validationCommand, passed: true });
    } catch (err) {
      checks.push({
        kind: "command",
        target: contract.validationCommand,
        passed: false,
        detail: err instanceof Error ? err.message.slice(0, 200) : "Command failed",
      });
    }
  }

  return {
    valid: checks.every((c) => c.passed),
    checks,
  };
}
```

### 2. Chain dispatch in `src/domains/dispatch/primitives.ts` (EXTEND)

Add `dispatchChain()` to the existing file alongside `runParallel()`.

```typescript
export interface ChainStep {
  task: string;          // supports $INPUT and $ORIGINAL substitution
  agent?: string;        // defaults to parent agent
  outputContract?: OutputContract;
}

export interface ChainResult {
  steps: Array<{
    stepIndex: number;
    agent: string;
    task: string;
    result: WorkerResult;
    validation?: ValidationResult;
    durationMs: number;
  }>;
  finalOutput: string;
  success: boolean;
  failedAtStep?: number;
}

const MAX_SUBSTITUTED_OUTPUT = 8000; // cap injected output to prevent prompt explosion

export async function dispatchChain(
  steps: ChainStep[],
  originalTask: string,
  defaultAgent: string,
  spawnFn: (task: string, agent: string) => Promise<WorkerResult>,
  validateFn?: (output: string, cwd: string, contract: OutputContract) => ValidationResult,
  signal?: AbortSignal,
): Promise<ChainResult> {
  const results: ChainResult["steps"] = [];
  let previousOutput = "";

  for (let i = 0; i < steps.length; i++) {
    if (signal?.aborted) {
      return { steps: results, finalOutput: previousOutput, success: false, failedAtStep: i };
    }

    const step = steps[i];
    const agent = step.agent ?? defaultAgent;
    const cappedPrevious = previousOutput.slice(0, MAX_SUBSTITUTED_OUTPUT);
    const cappedOriginal = originalTask.slice(0, MAX_SUBSTITUTED_OUTPUT);

    // Token substitution
    const task = step.task
      .replace(/\$INPUT/g, cappedPrevious)
      .replace(/\$ORIGINAL/g, cappedOriginal);

    const startTime = Date.now();
    const result = await spawnFn(task, agent);
    const durationMs = Date.now() - startTime;

    // Optional validation
    let validation: ValidationResult | undefined;
    if (step.outputContract && validateFn) {
      validation = validateFn(result.result, process.cwd(), step.outputContract);
    }

    results.push({ stepIndex: i, agent, task, result, validation, durationMs });

    if (result.exitCode !== 0 || result.error) {
      return { steps: results, finalOutput: result.result, success: false, failedAtStep: i };
    }

    if (validation && !validation.valid) {
      return { steps: results, finalOutput: result.result, success: false, failedAtStep: i };
    }

    previousOutput = result.result;
  }

  return { steps: results, finalOutput: previousOutput, success: true };
}
```

- Show last 5 dispatch rule matches
- Table format
